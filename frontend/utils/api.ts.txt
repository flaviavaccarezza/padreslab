const API_URL = process.env.EXPO_PUBLIC_BACKEND_URL;

export interface SituationInput {
  text: string;
  user_id?: string;
}

export interface ComoHablar {
  frases_sugeridas: string[];
  que_evitar: string[];
  tono_recomendado: string;
}

export interface SenalesDeAlerta {
  signos: string[];
  cuando_actuar: string;
}

export interface ProximosPasos {
  que_hacer_ahora: string;
  que_observar: string[];
  como_seguir: string;
}

export interface Reconexion {
  preguntas: string[];
  actividades: string[];
}

export interface SituationSections {
  que_podria_estar_pasando: string;
  como_hablar: ComoHablar;
  senales_de_alerta: SenalesDeAlerta;
  proximos_pasos: ProximosPasos;
  reconexion: Reconexion;
}

export interface SituationResponse {
  id: string;
  intensidad_emocional: 'baja' | 'media' | 'alta';
  riesgo_alto: boolean;
  mensaje_ayuda_profesional?: string;
  secciones: SituationSections;
  timestamp: string;
}

export interface RewriteInput {
  original_text: string;
  context?: string;
}

export interface RewriteResponse {
  versiones: string[];
}

export interface FrequentSituation {
  id: string;
  titulo: string;
  descripcion: string;
  emoji: string;
}

export interface HelpResource {
  tipo: string;
  nombre: string;
  telefono?: string;
  descripcion: string;
}

export const analyzeSituation = async (input: SituationInput): Promise<SituationResponse> => {
  const response = await fetch(`${API_URL}/api/analyze-situation`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(input),
  });

  if (!response.ok) {
    throw new Error('Error al analizar la situación');
  }

  return response.json();
};

export const rewriteMessage = async (input: RewriteInput): Promise<RewriteResponse> => {
  const response = await fetch(`${API_URL}/api/rewrite-message`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(input),
  });

  if (!response.ok) {
    throw new Error('Error al reescribir el mensaje');
  }

  return response.json();
};

export const getFrequentSituations = async (): Promise<FrequentSituation[]> => {
  const response = await fetch(`${API_URL}/api/frequent-situations`);

  if (!response.ok) {
    throw new Error('Error al obtener situaciones frecuentes');
  }

  return response.json();
};

export const getHelpResources = async (): Promise<HelpResource[]> => {
  const response = await fetch(`${API_URL}/api/help-resources`);

  if (!response.ok) {
    throw new Error('Error al obtener recursos de ayuda');
  }

  return response.json();
};
